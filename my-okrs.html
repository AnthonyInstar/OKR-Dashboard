function getProgressColor(target, actual) {

const t = parseInt(target, 10) || 0, a = parseInt(actual, 10) || 0, diff = a - t;

if (Math.abs(diff) <= 10) return '#f59e0b';

if (diff < -10) return '#ef4444';

return '#22c55e';

}


function getStatusLabel(target, actual) {

const t = parseInt(target, 10) || 0, a = parseInt(actual, 10) || 0;

if (t === 0 && a === 0) return 'No data';

const diff = a - t;

if (Math.abs(diff) <= 10) return 'Near';

if (diff < -10) return 'Behind';

return 'Ahead';

}


function getStatusClass(label) {

if (label === 'Ahead') return 'pill-green';

if (label === 'Near') return 'pill-orange';

if (label === 'Behind') return 'pill-red';

return 'pill-gray';

}


function updateStatusBubble(targetSelect, actualSelect, bubble) {

if (color === '#22c55e') bubble.classList.add('status-green');

else if (color === '#f59e0b') bubble.classList.add('status-orange');

else bubble.classList.add('status-red');

}

function pentagonPoints(cx, cy, r) {

const pts = [];

for (let i = 0; i < 5; i++) {

const angle = (-90 + i * 72) * Math.PI / 180;

pts.push(`${(cx + r * Math.cos(angle)).toFixed(1)},${(cy + r * Math.sin(angle)).toFixed(1)}`);

}

return pts.join(' ');

}


function trianglePoints(cx, cy, size) {

return `${cx},${cy-size} ${cx-size},${cy+size} ${cx+size},${cy+size}`;

}


function escapeXml(str) {

return String(str||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&apos;');

}


function escapeHtml(str) {

return String(str||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;').replace(/'/g,'&#039;');

}


function computeKrSummary(objIndex, krIndex, data) {

const obj = data[`obj${objIndex}`] || {}, kr = obj[`kr${krIndex}`] || {}, quarters = kr.quarters || {};

let count = 0, score = 0, missingCount = 0;

QUARTERS.forEach(q => {

const qData = quarters[q] || {};

const t = parseInt(safePercentValue(qData.target), 10), a = parseInt(safePercentValue(qData.actual), 10);

if (t === 0 && a === 0) { missingCount++; return; }

count++; score += (a - t);

});

if (!kr.label) return { label: 'Missing label', className: 'pill-gray', averageGap: null, missingCount };

if (count === 0) return { label: 'No data', className: 'pill-gray', averageGap: null, missingCount };

const avgGap = score / count, status = getStatusLabel(50, 50 + avgGap);

return { label: status, className: getStatusClass(status), averageGap: avgGap, missingCount };

}


function computeObjectiveSummary(objIndex, data) {

let active = 0, score = 0, hasMissing = false;

for (let krIndex = 1; krIndex <= OVERVIEW_KR_COUNT; krIndex++) {

const summary = computeKrSummary(objIndex, krIndex, data);

if (summary.label === 'Missing label' || summary.label === 'No data') hasMissing = true;

if (summary.averageGap !== null) { active++; score += summary.averageGap; }

}

if (active === 0) return { label: 'No data', color: 'rgba(255,255,255,0.18)' };

const avg = score / active;

if (avg > 10) return { label: hasMissing ? 'Ahead \'b7 partial' : 'Ahead', color: '#166534' };

if (avg < -10) return { label: hasMissing ? 'Behind \'b7 partial' : 'Behind', color: '#991b1b' };

return { label: hasMissing ? 'Mixed \'b7 partial' : 'Mixed', color: '#92400e' };

}


function getValidationMessage(objIndex, krIndex, data) {

const obj = data[`obj${objIndex}`] || {}, kr = obj[`kr${krIndex}`] || {}, issues = [];

if (!obj.title) issues.push('Objective title missing');

if (!kr.label) issues.push('KR label missing');

let hasQuarterData = false, missingQuarter = false;

QUARTERS.forEach(q => {

const qData = (kr.quarters || {})[q] || {};

const t = parseInt(safePercentValue(qData.target),10), a = parseInt(safePercentValue(qData.actual),10);

if (t !== 0 || a !== 0) hasQuarterData = true;

if (t === 0 && a === 0) missingQuarter = true;

});

if (!hasQuarterData) issues.push('No quarterly progress entered');

else if (missingQuarter) issues.push('Some quarters still blank');

return issues.join(' \'b7 ');

}
