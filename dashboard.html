  <script>

    // Sidebar person switching (simple visual switch, same layout reused)

    const personItems = document.querySelectorAll('.person-list li');

    const personTitle = document.getElementById('personTitle');


    personItems.forEach(li => {

      li.addEventListener('click', () => {

        personItems.forEach(i => i.classList.remove('active'));

        li.classList.add('active');

        personTitle.textContent = li.textContent;

        // Here you could later load person-specific OKR data

      });

    });


    // Traffic light behavior: click a light to turn that one "on", others off

    document.querySelectorAll('.traffic-light').forEach(lightBlock => {

      lightBlock.addEventListener('click', e => {

        const clicked = e.target.closest('.light');

        if (!clicked) return;


        const lights = lightBlock.querySelectorAll('.light');

        lights.forEach(l => l.classList.remove('on'));

        clicked.classList.add('on');

      });

    });


    // Make the whole task line toggle the checkbox

    document.querySelectorAll('.tasks li').forEach(li => {

      li.addEventListener('click', e => {

        const checkbox = li.querySelector('input[type="checkbox"]');

        if (e.target === checkbox) return;

        checkbox.checked = !checkbox.checked;

      });

    });

  </script>
